/**
 * @fileoverview
 * @namespace components
 * Questo file contiene il componente React `BuildingEnergyConsumption`, che permette di visualizzare
 * il consumo energetico totale per edificio in un dato intervallo di tempo. Il componente consente
 * all'utente di selezionare un intervallo di tempo, visualizzare i consumi separati per edificio e
 * esportare i dati in formato Excel. Inoltre, vengono evidenziati l'edificio con il costo più alto
 * e quello con il costo più basso.
 *
 * @module BuildingEnergyConsumption
 */

import React, { useState, useEffect } from "react";
import TimeRangePicker from "../common/TimeRangePicker";
import LoadingSpinner from "../common/LoadingSpinner";
import ErrorDisplay from "../common/ErrorDisplay";
import ResultCard from "../common/ResultCard";
import { fetchTotalConsumptionByBuilding } from "../../utils/consumiAPI";
import { exportQueryDataToExcel } from "../../utils/exportAPI";

import useTimeRange from "../../hooks/useTimeRange";

/**
 * Componente per visualizzare il consumo energetico totale per edificio.
 * Permette di selezionare un intervallo di tempo, visualizzare i consumi per edificio
 * e esportare i dati in formato Excel.
 *
 * @component
 * @example
 * return (
 *   <BuildingEnergyConsumption />
 * );
 *
 * @returns {JSX.Element} Componente React che visualizza i consumi energetici per edificio.
 */
const BuildingEnergyConsumption = () => {
  const { startDate, endDate, activeRange, handleSelectRange } =
    useTimeRange("week"); // Usa l'hook

  const [consumptionData, setConsumptionData] = useState(null);
  const [error, setError] = useState(null);
  const [isLoading, setIsLoading] = useState(false);

  /**
   * Funzione per recuperare i consumi energetici totali per edificio
   * nel periodo selezionato.
   *
   * @async
   * @function
   * @returns {void}
   */
  const fetchConsumptionData = async () => {
    try {
      setIsLoading(true);
      setError(null);
      const data = await fetchTotalConsumptionByBuilding(startDate, endDate);
      setConsumptionData(data);
    } catch (err) {
      console.error("Errore durante il recupero dei dati:", err.message);
      setError("Impossibile recuperare i consumi. Verifica le date.");
    } finally {
      setIsLoading(false);
    }
  };

  /**
   * Funzione per trovare l'edificio con il costo totale più alto.
   *
   * @param {Array} consumptionByBuilding Array contenente i dati di consumo per edificio.
   * @returns {Object|null} Edificio con il costo più alto, o null se i dati non sono disponibili.
   */
  const getMaxCostBuilding = (consumptionByBuilding) => {
    if (!consumptionByBuilding || consumptionByBuilding.length === 0)
      return null;
    return consumptionByBuilding.reduce((maxBuilding, currentBuilding) => {
      const currentCost = currentBuilding.consumption * 0.12; // Calcola il costo (prezzo per unità)
      const maxCost = maxBuilding.consumption * 0.12;
      return currentCost > maxCost ? currentBuilding : maxBuilding;
    });
  };

  /**
   * Funzione per trovare l'edificio con il costo totale più basso.
   *
   * @param {Array} consumptionByBuilding Array contenente i dati di consumo per edificio.
   * @returns {Object|null} Edificio con il costo più basso, o null se i dati non sono disponibili.
   */
  const getMinCostBuilding = (consumptionByBuilding) => {
    if (!consumptionByBuilding || consumptionByBuilding.length === 0)
      return null;
    return consumptionByBuilding.reduce((minBuilding, currentBuilding) => {
      const currentCost = currentBuilding.consumption * 0.12; // Calcola il costo (prezzo per unità)
      const minCost = minBuilding.consumption * 0.12;
      return currentCost < minCost ? currentBuilding : minBuilding;
    });
  };

  const maxCostBuilding = consumptionData
    ? getMaxCostBuilding(consumptionData.consumptionByBuilding)
    : null;

  const minCostBuilding = consumptionData
    ? getMinCostBuilding(consumptionData.consumptionByBuilding)
    : null;

  /**
   * Funzione per esportare i dati di consumo in formato Excel.
   *
   * @async
   * @function
   * @returns {void}
   */
  const handleExportToExcel = async () => {
    try {
      await exportQueryDataToExcel(
        "BuildingEnergyConsumption",
        "energyDataService",
        "getTotalConsumptionByBuilding",
        { startDate, endDate }
      );
    } catch (error) {
      console.error("Errore durante l'esportazione:", error.message);
      toast.error("Errore durante l'esportazione.");
    }
  };

  // Esegui il recupero dei dati solo quando startDate e endDate cambiano effettivamente
  useEffect(() => {
    if (startDate && endDate) {
      fetchConsumptionData();
    }
  }, [startDate, endDate]);

  return (
    <div className="energy-consumption">
      {/* TimeRangePicker per la selezione del periodo */}
      <TimeRangePicker
        onSelectRange={handleSelectRange}
        isLoading={isLoading}
        defaultRange={activeRange} // Passa il range attivo al TimeRangePicker
      />

      <button
        className="mt-4 px-4 py-2 bg-green-500 text-white rounded"
        onClick={handleExportToExcel}
      >
        Esporta in Excel
      </button>

      {isLoading && <LoadingSpinner />}

      <div className="results mt-8">
        {error && <ErrorDisplay message={error} />}
        {consumptionData ? (
          <div className="result-card p-6 bg-gray-100 rounded-lg">
            <p>
              <strong>Data Inizio:</strong>{" "}
              {new Date(consumptionData.startDate).toLocaleString()}
            </p>
            <p>
              <strong>Data Fine:</strong>{" "}
              {new Date(consumptionData.endDate).toLocaleString()}
            </p>
            <div className="mt-4">
              <strong>Consumi per Edificio:</strong>
              <div className="building-consumptions mt-2 grid grid-cols-2 sm:grid-cols-3 md:grid-cols-4 gap-4">
                {consumptionData.consumptionByBuilding.map(
                  (building, index) => {
                    const cost = (building.consumption * 0.12).toFixed(2);
                    const isHighestCost =
                      maxCostBuilding && building === maxCostBuilding;
                    const isLowestCost =
                      minCostBuilding && building === minCostBuilding;

                    return (
                      <ResultCard
                        key={index}
                        building={building.building}
                        consumption={building.consumption.toFixed(2)}
                        unit={building.unit}
                        cost={cost}
                        isHighestCost={isHighestCost}
                        isLowestCost={isLowestCost}
                      />
                    );
                  }
                )}
              </div>
            </div>
          </div>
        ) : (
          <p>Nessun dato disponibile.</p>
        )}
      </div>
    </div>
  );
};

export default BuildingEnergyConsumption;
